<h2>أدخل رقم العضوية</h2>

<input type="text" id="id">
<button onclick="search()">بحث</button>

<div id="result"></div>

<script>
const data = [
  {id:"143", name:"سيام جابيش", total:"50000000", paid:"2000000", remaining:"48000000"},
  {id:"200", name:"محمد عساف", total:"2000000", paid:"500000", remaining:"1500000"}
];

function search(){
  let val = document.getElementById("id").value;
  let m = data.find(x=>x.id==val);

  if(m){
    document.getElementById("result").innerHTML =
    "الاسم: "+m.name+"<br>الإجمالي: "+m.total+
    "<br>المدفوع: "+m.paid+
    "<br>المتبقي: "+m.remaining;
  } else {
    document.getElementById("result").innerHTML = "لا يوجد";
  }
}
</script>
